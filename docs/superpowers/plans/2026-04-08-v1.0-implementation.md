```python
@staticmethod
def _strip_inline_comment(text: str) -> tuple[str, str | None]:
    """Strip trailing inline comment from a line.

    Returns (content, comment_text) where comment_text is None if no comment.
    An inline comment is ' //' (space + double-slash) not inside a code span.
    """
    # Track whether we're inside backticks
    in_code = False
    i = 0
    while i < len(text):
        ch = text[i]
        if ch == '\\' and i + 1 < len(text):
            i += 2  # skip escape
            continue
        if ch == '`':
            in_code = not in_code
        elif ch == '/' and not in_code:
            if i > 0 and text[i - 1] in (' ', '\t') and i + 1 < len(text) and text[i + 1] == '/':
                return text[:i].rstrip(), text[i + 2:]
        i += 1
    return text, None
```

```python
def _normalize_directive(
    self,
    block: BlockDirective,
    pending_anchor: str | None,
    *,
    include_stack: tuple[Path, ...] = (),
    include_depth: int = 0,
) -> tuple[NormalizedBlock | None, str | None]:
    if block.name == "include":
        src = cast(str, block.attrs.get("src"))
        # Resolve path relative to current file
        target = (self._current_path.parent / src).resolve()

        # Circular detection
        if target in include_stack:
            raise ValidationFailure(
                "circular_include",
                f"Circular include: {' -> '.join(str(p) for p in include_stack)} -> {target}",
            )

        # Depth check
        if include_depth >= 10:
            raise ValidationFailure(
                "include_depth_exceeded",
                f"Include depth exceeds 10 levels",
            )

        # Read, parse, validate, normalize the target
        source = target.read_text(encoding="utf-8")
        parser = ReferenceParser(self.registry)
        doc = parser.parse_document(source, target)
        # Validate (reuse config)
        validator = ReferenceValidator(self.registry)
        validator.validate(doc, config=self._config)
        # Normalize recursively with include tracking
        child_normalizer = Normalizer(self.registry)
        child_normalizer._current_path = target
        child_normalizer._config = self._config
        child_normalizer.slug_counts = self.slug_counts  # share slug state
        child_normalizer.note_counter = self.note_counter
        child_doc = child_normalizer.normalize(
            doc,
            include_stack=include_stack + (target,),
            include_depth=include_depth + 1,
        )
        # Update shared state
        self.note_counter = child_normalizer.note_counter
        self.notes.extend(child_doc.notes)
        # Return inlined blocks (meta discarded)
        return child_doc.blocks, pending_anchor  # special: returns list
```

Note: this is pseudocode. The actual implementation will need to handle the `_normalize_blocks` return type since `include` returns multiple blocks, not one. Refactor `_normalize_directive` to return `list[NormalizedBlock]` or handle inline expansion in `_normalize_blocks`.

```python
def _parse_unordered_list(self, lines: list[str], index: int, *, indent: int = 0) -> tuple[UnorderedList, int]:

    items: list[ListItem] = []
    while index < len(lines):
        line = lines[index]
        # Check indentation level
        stripped = line.lstrip(' ')
        line_indent = len(line) - len(stripped)
        if '\t' in line[:line_indent]:
            raise ParseFailure("invalid_list_indent", "Tabs not allowed in list indentation", line=index + 1)
        if line_indent < indent:
            break  # dedent ends this list
        if line_indent > indent:
            # Could be continuation or nested list
            ...
        trimmed = stripped
        if not trimmed.startswith("- "):
            break
        # Parse this item
        content_text = trimmed[2:]
        children = self._parse_inline_line(content_text, line=index + 1)
        index += 1
        # Collect nested blocks (sub-lists, continuation paragraphs)
        blocks, index = self._parse_list_item_body(lines, index, item_indent=indent + 2)
```

```python
def _parse_list_item_body(self, lines: list[str], index: int, *, item_indent: int) -> tuple[list[BlockNode], int]:
    """Parse continuation blocks for a list item at the given indent."""
    blocks: list[BlockNode] = []
    while index < len(lines):
        if self._is_blank(lines[index]):
            # Blank line: check if next line is indented (continuation)
            next_idx = index + 1
            while next_idx < len(lines) and self._is_blank(lines[next_idx]):
                next_idx += 1
            if next_idx >= len(lines):
                break
            next_line = lines[next_idx]
            next_indent = len(next_line) - len(next_line.lstrip(' '))
            if next_indent < item_indent:
                break  # dedent after blank = end of item
            index = next_idx
            continue
        line = lines[index]
        line_indent = len(line) - len(line.lstrip(' '))
        if line_indent < item_indent:
            break
        # Parse the indented content as a block (strip indent first)
        dedented = line[item_indent:]
```
